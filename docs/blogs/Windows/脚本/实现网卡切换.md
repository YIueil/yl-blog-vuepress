# 实现网卡切换
>企业开发中经常需要内外网切换，之前使用物理切换器进行网络切换，可行但是占用地方，改用脚本切换实现。

修改方法：使用前更名两个网卡名称。
```powershell
# ================= Config =================
$AdapterEth  = "YIueilEth"
$AdapterWifi = "YIueilWLAN"
# ==========================================

# Auto elevate to admin if not already
If (!([Security.Principal.WindowsPrincipal][Security.Principal.WindowsIdentity]::GetCurrent()).IsInRole([Security.Principal.WindowsBuiltInRole] "Administrator")) {
    Start-Process powershell.exe "-NoProfile -ExecutionPolicy Bypass -File `"$PSCommandPath`"" -Verb RunAs
    Exit
}

# Get adapter info
$ethAdapter  = Get-NetAdapter -Name $AdapterEth -ErrorAction SilentlyContinue
$wifiAdapter = Get-NetAdapter -Name $AdapterWifi -ErrorAction SilentlyContinue

# Check if adapters exist
if (!$ethAdapter -or !$wifiAdapter) {
    Write-Host "ERROR: Adapter not found. Please check names in script." -ForegroundColor Red
    Write-Host "Press any key to exit..."
    $null = $Host.UI.RawUI.ReadKey("NoEcho,IncludeKeyDown")
    Exit
}

# Use AdminStatus to check Enabled/Disabled (Up=Enabled, Down=Disabled)
$ethEnabled  = ($ethAdapter.AdminStatus -eq "Up")
$wifiEnabled = ($wifiAdapter.AdminStatus -eq "Up")

# Display current status
Write-Host "Current Status:" -ForegroundColor Gray
Write-Host "  Ethernet : $($ethAdapter.Status) (Enabled: $ethEnabled)" -ForegroundColor Gray
Write-Host "  WLAN     : $($wifiAdapter.Status) (Enabled: $wifiEnabled)" -ForegroundColor Gray
Write-Host ""

# Toggle Logic
if ($ethEnabled -and !$wifiEnabled) {
    # Currently Ethernet, switch to WLAN
    Write-Host ">>> Switching to WLAN..." -ForegroundColor Cyan
    Enable-NetAdapter -Name $AdapterWifi -Confirm:$false -ErrorAction SilentlyContinue
    Start-Sleep -Seconds 1
    Disable-NetAdapter -Name $AdapterEth -Confirm:$false -ErrorAction SilentlyContinue
    Write-Host ">>> Done! Now using WiFi." -ForegroundColor Green
}
elseif ($wifiEnabled -and !$ethEnabled) {
    # Currently WLAN, switch to Ethernet
    Write-Host ">>> Switching to Ethernet..." -ForegroundColor Yellow
    Enable-NetAdapter -Name $AdapterEth -Confirm:$false -ErrorAction SilentlyContinue
    Start-Sleep -Seconds 1
    Disable-NetAdapter -Name $AdapterWifi -Confirm:$false -ErrorAction SilentlyContinue
    Write-Host ">>> Done! Now using Ethernet." -ForegroundColor Green
}
elseif ($ethEnabled -and $wifiEnabled) {
    # Both enabled, disable WiFi (Priority to Ethernet)
    Write-Host ">>> Both enabled. Disabling WiFi..." -ForegroundColor Yellow
    Disable-NetAdapter -Name $AdapterWifi -Confirm:$false -ErrorAction SilentlyContinue
    Write-Host ">>> Done! Now using Ethernet." -ForegroundColor Green
}
else {
    # Both disabled, enable Ethernet
    Write-Host ">>> Both disabled. Enabling Ethernet..." -ForegroundColor Red
    Enable-NetAdapter -Name $AdapterEth -Confirm:$false -ErrorAction SilentlyContinue
    Write-Host ">>> Done! Now using Ethernet." -ForegroundColor Green
}

# Wait for changes to apply
Start-Sleep -Seconds 2

# Refresh adapter info and display AFTER status
$ethAdapter  = Get-NetAdapter -Name $AdapterEth -ErrorAction SilentlyContinue
$wifiAdapter = Get-NetAdapter -Name $AdapterWifi -ErrorAction SilentlyContinue

$ethEnabled  = ($ethAdapter.AdminStatus -eq "Up")
$wifiEnabled = ($wifiAdapter.AdminStatus -eq "Up")

Write-Host "=== AFTER ===" -ForegroundColor Green
Write-Host "  Ethernet : $($ethAdapter.Status) (Enabled: $ethEnabled)" -ForegroundColor Gray
Write-Host "  WLAN     : $($wifiAdapter.Status) (Enabled: $wifiEnabled)" -ForegroundColor Gray
Write-Host ""
Write-Host ">>> Switch Complete!" -ForegroundColor Green
Write-Host ""
Write-Host "Press any key to close..." -ForegroundColor Gray
$null = $Host.UI.RawUI.ReadKey("NoEcho,IncludeKeyDown")
```
